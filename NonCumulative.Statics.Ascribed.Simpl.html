∷-cong-simp : ∀ {i} →
              ⊢ Γ ≈ Δ →
              Γ ⊢ T ≈ T′ ∶[ 1 + i ] Se i →
              ----------------
              ⊢ (T ↙ i) ∷ Γ ≈ (T′ ↙ i) ∷ Δ
∷-cong-simp Γ≈Δ T≈T′
  with presup-≈ T≈T′
...  | _ , ⊢T , ⊢T′ , _ = ∷-cong Γ≈Δ ⊢T (ctxeq-tm Γ≈Δ ⊢T′) T≈T′ (ctxeq-≈ Γ≈Δ T≈T′)
